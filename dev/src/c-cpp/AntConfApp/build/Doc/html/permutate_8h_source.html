50  inline bool next_combination(const Iterator first, Iterator k, const Iterator last)
51  {
52  /* Credits: Thomas Draper */
53  if ((first == last) || (first == k) || (last == k))
54  return false;
55  Iterator itr1 = first;
56  Iterator itr2 = last;
57  ++itr1;
58  if (last == itr1)
59  return false;
60  itr1 = last;
61  --itr1;
62  itr1 = k;
63  --itr2;
64  while (first != itr1)
65  {
66  if (*--itr1 < *itr2)
67  {
68  Iterator j = k;
69  while (!(*itr1 < *j)) ++j;
70  std::iter_swap(itr1,j);
71  ++itr1;
72  ++j;
73  itr2 = k;
74  std::rotate(itr1,j,last);
75  while (last != j)
76  {
77  ++j;
78  ++itr2;
79  }
80  std::rotate(k,itr2,last);
81  return true;
82  }
83  }
84  std::rotate(first,k,last);
85  return false;
86  }
